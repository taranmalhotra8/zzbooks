\usepackage{tikz}
\usetikzlibrary{calc}
\makeatletter
\renewcommand{\maketitle}{%
  \begin{titlepage}
    \begin{tikzpicture}[remember picture, overlay]
      % Full-page gradient background
      \fill[left color=zopdev-cover-start, right color=zopdev-cover-mid]
        (current page.south west) rectangle (current page.north east);
      % Gradient overlay from mid to end
      \fill[left color=zopdev-cover-mid, right color=zopdev-cover-end, opacity=0.6]
        (current page.south west) rectangle (current page.north east);
      % Grid pattern
      \foreach \i in {0,1,...,15} {
        \draw[white, opacity=0.03, line width=0.5pt]
          ($(current page.south west) + (\i*1.2cm, 0)$) -- ++(0, \paperheight);
        \draw[white, opacity=0.03, line width=0.5pt]
          ($(current page.south west) + (0, \i*1.5cm)$) -- ++(\paperwidth, 0);
      }
      % Decorative accent circles
      \fill[zopdev-primary, opacity=0.08]
        ($(current page.north east) + (-3cm, -4cm)$) circle (5cm);
      \fill[zopdev-indigo, opacity=0.06]
        ($(current page.south west) + (2cm, 3cm)$) circle (4cm);
    \end{tikzpicture}
    % Content
    \vspace*{4cm}
    \begin{flushleft}
      {\fontsize{11}{14}\selectfont\color{white}\fontseries{sb}\selectfont\MakeUppercase{Zopdev}\par}
      \vspace{0.3cm}
      {\color{white}\rule{40pt}{2pt}\par}
      \vspace{2cm}
      {\fontsize{38}{44}\selectfont\bfseries\color{white}\@title\par}
      \vspace{0.75cm}
      \ifdefined\@subtitle{\fontsize{16}{22}\selectfont\color{white!80}\@subtitle\par}\fi
      \vspace{3cm}
      {\fontsize{12}{16}\selectfont\color{white!70}\@author\par}
      \vspace{0.5cm}
      {\fontsize{11}{14}\selectfont\color{white!50}\@date\par}
    \end{flushleft}
    \vfill
    \begin{flushleft}
      {\color{white!30}\rule{3cm}{0.5pt}}
    \end{flushleft}
  \end{titlepage}
}
